// game.js const dictionary = [ "ability", "absence", "academy", "account", "accused", "achieve", "acquire", "address", "advance", "adverse", "advised", "adviser", "against", "airline", "airport", "alleged", "already", "amateur", "amazing", "analyst" ]; let targetWord = ""; let currentIndex = 0; let score = 0; let timeLeft = 35; let recognition; let timerInterval; let mode = "mic"; let usedWords = new Set(); const startBtn = document.getElementById('startBtn'); const micModeBtn = document.getElementById('micModeBtn'); const typeModeBtn = document.getElementById('typeModeBtn'); const toggleDarkModeBtn = document.getElementById('toggleDarkModeBtn'); const modeButtons = document.getElementById('modeButtons'); const targetDiv = document.getElementById('target'); const timerDisplay = document.getElementById('timeDisplay'); const bonusDiv = document.getElementById('bonusAnimation'); const logDiv = document.getElementById('log'); const scoreBoard = document.getElementById('scoreBoard'); const inputArea = document.getElementById('inputArea'); const typedInput = document.getElementById('typedInput'); const submitBtn = document.getElementById('submitBtn'); const usedWordsUl = document.getElementById('usedWordsUl'); const successSound = document.getElementById('successSound'); const errorSound = document.getElementById('errorSound'); const bonusSound = document.getElementById('bonusSound'); toggleDarkModeBtn.addEventListener('click', () => { document.body.classList.toggle('dark-mode'); }); function randomWord(length = 7) { const words = dictionary.filter(w => w.length === length); return words[Math.floor(Math.random() * words.length)] || "awesome"; } function displayTargetWord() { targetDiv.innerHTML = ""; for (let i = 0; i < targetWord.length; i++) { const span = document.createElement('span'); span.textContent = targetWord[i].toUpperCase(); span.className = 'letter'; if (i === currentIndex) span.classList.add('active'); targetDiv.appendChild(span); } } function updateTimer() { timerDisplay.textContent = `${timeLeft.toFixed(1)}s`; } function logMessage(message, type = "info") { const p = document.createElement('p'); p.textContent = message; p.className = `log-${type}`; logDiv.appendChild(p); logDiv.scrollTop = logDiv.scrollHeight; } function showBonusAnimation() { bonusDiv.style.display = 'block'; bonusDiv.style.animation = 'none'; void bonusDiv.offsetWidth; bonusDiv.style.animation = 'floatUp 1.5s forwards'; setTimeout(() => { bonusDiv.style.display = 'none'; }, 1500); } function updateUsedWords(word) { const li = document.createElement('li'); li.textContent = word; usedWordsUl.appendChild(li); } function startTimer() { timerInterval = setInterval(() => { timeLeft -= 0.1; updateTimer(); if (timeLeft <= 0) { endGame(); } }, 100); } function endGame() { clearInterval(timerInterval); if (recognition) recognition.stop(); timerDisplay.textContent = "Time's up!"; scoreBoard.textContent = `Final Score: ${score} points`; startBtn.style.display = "none"; modeButtons.style.display = "block"; inputArea.style.display = "none"; } function nextLetter() { currentIndex++; if (currentIndex >= targetWord.length) { targetWord = randomWord(); currentIndex = 0; logMessage("New word generated!", "info"); } displayTargetWord(); } function validateWord(spokenWord) { const word = spokenWord.toLowerCase(); const currentLetter = targetWord[currentIndex].toLowerCase(); if (usedWords.has(word)) { errorSound.play(); logMessage(`Word already used: ${spokenWord}`, "error"); return; } if (word && word[0] === currentLetter) { usedWords.add(word); updateUsedWords(word); let points = word.length; score += points; scoreBoard.classList.add('bounce'); setTimeout(() => scoreBoard.classList.remove('bounce'), 400); successSound.play(); if (word.length >= 10) { timeLeft += 1.5; showBonusAnimation(); bonusSound.play(); logMessage(`+1.5s bonus for long word!`, "success"); } logMessage(`Correct! ${spokenWord} (+${points} pts)`, "success"); nextLetter(); } else { errorSound.play(); logMessage(`Word doesn't match current letter: ${spokenWord}`, "error"); } } function startRecognition() { window.SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; recognition = new SpeechRecognition(); recognition.lang = 'en-US'; recognition.interimResults = false; recognition.onresult = (e) => { const spoken = e.results[0][0].transcript.trim(); logMessage(`Heard: ${spoken}`, "info"); validateWord(spoken); }; recognition.onerror = (e) => { logMessage(`Mic error: ${e.error}`, "error"); }; recognition.onend = () => { if (timeLeft > 0) recognition.start(); }; recognition.start(); } startBtn.addEventListener('click', () => { score = 0; timeLeft = 35; usedWords.clear(); usedWordsUl.innerHTML = ""; logDiv.innerHTML = ""; scoreBoard.textContent = ""; targetWord = randomWord(); currentIndex = 0; displayTargetWord(); updateTimer(); startTimer(); if (mode === "mic") { startRecognition(); inputArea.style.display = "none"; } else { inputArea.style.display = "block"; typedInput.focus(); } startBtn.style.display = "none"; modeButtons.style.display = "none"; }); micModeBtn.addEventListener('click', () => { mode = "mic"; startBtn.style.display = "inline-block"; }); typeModeBtn.addEventListener('click', () => { mode = "type"; startBtn.style.display = "inline-block"; }); submitBtn.addEventListener('click', () => { const typedWord = typedInput.value.trim(); if (typedWord) { validateWord(typedWord); typedInput.value = ""; } }); typedInput.addEventListener('keypress', (e) => { if (e.key === "Enter") { submitBtn.click(); } });
